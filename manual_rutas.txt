planeta.routes 

router.get('/', authenticate, allowRoles('admin', 'astro'), planetaController.getAllplaneta); // Vista de todos los planetas (más sus lunas asociadas)
router.get('/:idPlanet', authenticate, allowRoles('admin', 'astro'), planetaController.getplanetaByidPlanet); // Vista de un planeta específico (más sus lunas asociadas)
router.post('/', authenticate, allowRoles('admin'), planetaController.createplaneta); // Crear un planeta
router.patch('/:idPlanet/luna/:idLuna', authenticate, allowRoles('admin'), planetaController.updatelunaRelation); // Cambiar el idPlanet de una luna
router.delete('/:idPlanet', authenticate, allowRoles('admin'), planetaController.deleteplaneta); // Eliminar permanentemente un planeta (junto con sus lunas asociadas)
router.patch('/:idPlanet/soft-delete', authenticate, allowRoles('admin'), planetaController.softDeleteplaneta); // Ocultar un planeta (soft-delete)
router.patch('/:idPlanet/restore', authenticate, allowRoles('admin'), planetaController.restoreplaneta); // Restaurar un planeta (soft-delete NULL)
router.put('/:idPlanet', authenticate, allowRoles('admin'), planetaController.updateplaneta); // Modificar la información de un planeta
